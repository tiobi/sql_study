UNDEFINE N1
UNDEFINE N2
ACCEPT N1 PROMPT 'ENTER N1: ';
ACCEPT N2 PROMPT 'ENTER N2: ';

WITH ND AS (SELECT &N1 AS N1, &N2 AS N2
                FROM DUAL)
                
SELECT N1, N2, (SELECT &N1 AS N1, &N2 AS N2 FROM DUAL) AS MIN
    FROM ND
    WHERE MOD(N1, LEVEL) = 0
    AND MOD(N2, LEVEL) = 0
    CONNECT BY LEVEL <= N2;
