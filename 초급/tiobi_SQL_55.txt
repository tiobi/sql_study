SELECT ROW_NUMBER() OVER (ORDER BY sal DESC) AS ROW_NUM,
       empno, ename, sal, RANK() OVER (ORDER BY sal DESC) AS RANK,
       DENSE_RANK() OVER (ORDER BY sal DESC) AS DENSE_RANK
                          
    FROM emp
    WHERE deptno = 20;
